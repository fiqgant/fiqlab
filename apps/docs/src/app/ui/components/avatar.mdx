## Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from '@tszhong0411/ui'
```

```tsx
<Avatar>
  <AvatarImage src='https://github.com/tszhong0411.png' alt='@tszhong0411' />
  <AvatarFallback>TH</AvatarFallback>
</Avatar>
```
